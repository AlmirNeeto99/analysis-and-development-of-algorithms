def unique_Naive(Lista):
    list_len = len(Lista)
    for i in range(0, list_len - 1):
        for j in range(i + 1, list_len):
            if Lista[i] == Lista[j]:
                return False
    return True


def unique_Sort(Lista):
    Lista.sort()
    for i in range(0, len(Lista) - 1):
        if Lista[i] == Lista[i + 1]:
            return False
    return True


def unique_Hash(Lista):
    hash = set()
    for item in Lista:
        if item not in hash:
            hash.add(item)
        else:
            return False
    return True
